Quick Start
1

Create .scdev/config.yaml in your project:

.scdev/config.yamlversion: 1
name: my-project
domain: my-project.${DOMAIN}

shared:
  router: true

services:
  app:
    image: php:8.3-apache
    volumes:
      - .:/var/www/html
    routing:
      protocol: http
      port: 80
2

Start everything:

$ scdev start
3

Open https://my-project.${DOMAIN} in your browser.
